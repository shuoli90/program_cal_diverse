def extract_arguments(fh):
    # Read the line and split it into individual components
    line = fh.readline().strip().split()
    
    # Convert the components to integers
    N = int(line[0])
    A = int(line[1])
    B = int(line[2])
    C = int(line[3])
    D = int(line[4])
    
    # Return as a tuple
    return N, A, B, C, D
