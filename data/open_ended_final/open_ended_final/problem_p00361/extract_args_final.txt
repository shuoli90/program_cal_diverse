
def extract_arguments(fh):
    first_line = fh.readline().strip().split()
    N = int(first_line[0])
    M = int(first_line[1])
    roads = []
    for _ in range(M):
        s, t = map(int, fh.readline().strip().split())
        roads.append((s, t))
    return N, M, roads
