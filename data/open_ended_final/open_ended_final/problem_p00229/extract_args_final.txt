
def extract_arguments(fh):
    datasets = []
    while True:
        line = fh.readline().strip()
        if line == "0 0 0 0 0 0":
            break
        b, r, g, c, s, t = map(int, line.split())
        datasets.append((b, r, g, c, s, t))
    return datasets
